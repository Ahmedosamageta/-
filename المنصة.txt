let currentLang = null;
let filteredCodes = [];

function openModal(lang) {
  currentLang = lang;
  const d = data[lang];
  filteredCodes = [...d.codes];

  document.getElementById('modalTitle').innerHTML = `<span style="color:${d.color}">${d.title}</span>`;
  document.getElementById('searchInput').value = '';
  document.getElementById('codeDetail').innerHTML = `<div style="text-align:center; padding:60px 20px; color:var(--muted)"><div style="font-size:50px;margin-bottom:20px">??</div><div style="font-family:'Space Mono',monospace;font-size:13px">اختر كوداً من القائمة</div></div>`;
  renderCodesList();
  document.getElementById('codesModal').classList.add('active');
}

function renderCodesList() {
  const list = document.getElementById('codesList');
  const d = data[currentLang];
  list.innerHTML = filteredCodes.map((item, i) => `
    <div class="code-item" onclick="showCodeDetail(${data[currentLang].codes.indexOf(item)})">
      <div class="code-item-num">#${String(data[currentLang].codes.indexOf(item)+1).padStart(2,'0')}</div>
      <div class="code-item-tag" style="color:${d.color}">${item.tag}</div>
      <div class="code-item-desc">${item.desc}</div>
    </div>
  `).join('');
}

function filterCodes() {
  const q = document.getElementById('searchInput').value.toLowerCase();
  filteredCodes = data[currentLang].codes.filter(c =>
    c.tag.toLowerCase().includes(q) || c.desc.toLowerCase().includes(q)
  );
  renderCodesList();
}

function showCodeDetail(idx) {
  const item = data[currentLang].codes[idx];
  const d = data[currentLang];
  const langName = { html: 'HTML', css: 'CSS', js: 'JavaScript' }[currentLang];

  // Mark active
  document.querySelectorAll('.code-item').forEach(el => el.classList.remove('active'));

  let codeExample = '';
  if (currentLang === 'html') {
    codeExample = `<!-- HTML Code Example -->\n<${item.tag.replace(/[<>]/g,'').split(' ')[0]}>\n  المحتوى هنا\n</${item.tag.replace(/[<>]/g,'').split(' ')[0]}>`;
  } else if (currentLang === 'css') {
    codeExample = `/* CSS Property */\n.element {\n  ${item.tag}: value;\n}`;
  } else {
    codeExample = `// JavaScript\n// ${item.tag}\nconsole.log("${item.tag}");`;
  }

  document.getElementById('codeDetail').innerHTML = `
    <div class="detail-title" style="color:${d.color}">${item.tag}</div>
    <div class="detail-code-block" data-lang="${langName}">${escapeHTML(codeExample)}</div>
    <div class="detail-description">
      <h4>?? الشرح</h4>
      ${item.desc}
    </div>
  `;
}

function escapeHTML(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function closeModal() {
  document.getElementById('codesModal').classList.remove('active');
}

let activeTab = 'html';
function switchTab(tab, btn) {
  editorCode[activeTab] = document.getElementById('codeEditorArea').value;
  activeTab = tab;
  document.getElementById('codeEditorArea').value = editorCode[tab];
  document.querySelectorAll('.editor-tab').forEach(t => t.classList.remove('active'));
  btn.classList.add('active');
}

function runCode() {
  editorCode[activeTab] = document.getElementById('codeEditorArea').value;

  const combined = `<!DOCTYPE html><html><head><style>${editorCode.css}</style></head><body>${editorCode.html}<script>${editorCode.js}<\/script></body></html>`;
  frame.srcdoc = combined;
}
